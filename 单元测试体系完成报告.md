```python
async def test_execute_multiple_tasks(self, executor):
    """测试多任务并发执行"""
    async def async_task(x):
        await asyncio.sleep(0.1)
        return x ** 2
    
    tasks = [AsyncTask(f"task_{i}", async_task, i) for i in range(5)]
    results = await executor.execute_tasks(tasks)

    assert results == expected
```

```python
def test_retry_policy(self, handler):
    """测试重试策略"""
    policy = RetryPolicy(max_attempts=3, delay=0.1, backoff_factor=2.0)
    
    attempt_count = 0
    def failing_function():
        nonlocal attempt_count
        attempt_count += 1
        if attempt_count < 3:
            raise NetworkError("临时网络错误")
        return "成功"
    
    result = handler.execute_with_retry(failing_function, policy)
    assert result == "成功"
    assert attempt_count == 3
```

## 测试最佳实践

### 1. 测试命名规范
```python
def test_[功能]_[场景]_[预期结果](self):
    """测试[具体功能描述]"""
    # Given - 准备测试数据
    # When - 执行被测试的操作
    # Then - 验证结果
```

### 2. 断言策略
```python
# 使用具体的断言
assert result.success is True
assert len(result.images) == 3
assert result.message == "解析成功"

# 避免模糊的断言
assert result  # 不够具体
assert result.images  # 不够明确
```

### 3. 测试隔离原则
- 每个测试独立运行
- 使用临时目录和文件
- 清理测试产生的副作用
- 避免测试间的依赖关系
